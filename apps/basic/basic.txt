FUNCTION Max(BYVAL a AS INTEGER, BYVAL b AS INTEGER) AS INTEGER
	IF a > b THEN
		RETURN a
	ELSE
		RETURN b
	END IF
END FUNCTION

SUB Sum100(BYREF ref AS INTEGER)
	DIM i
	FOR i IN 1 TO 100 STEP 1
		i = i + 1
	NEXT
END SUB
